43@contextmanager 

44def _check_require_version(namespace, stacklevel): 

45 """A context manager which tries to give helpful warnings 

46 about missing gi.require_version() which could potentially 

47 break code if only an older version than expected is installed 

48 or a new version gets introduced. 

49 

50 :: 

51 

52 with _check_require_version("Gtk", stacklevel): 

53 load_namespace_and_overrides() 

54 """ 

55 

56 was_loaded = repository.is_registered(namespace) 

57 

58 yield 

59 

60 if was_loaded: 

61 # it was loaded before by another import which depended on this 

62 # namespace or by C code like libpeas 

63 return 

64 

65 if namespace in ("GLib", "GObject", "Gio"): 

66 # part of glib (we have bigger problems if versions change there) 

67 return 

68 

69 if gi.get_required_version(namespace) is not None: 

70 # the version was forced using require_version() 

71 return 

72 

73 version = repository.get_version(namespace) 

74 warnings.warn( 

75 "%(namespace)s was imported without specifying a version first. " 

76 "Use gi.require_version('%(namespace)s', '%(version)s') before " 

77 "import to ensure that the right version gets loaded." 

78 % {"namespace": namespace, "version": version}, 

79 PyGIWarning, stacklevel=stacklevel)

82def get_import_stacklevel(import_hook): 

83 """Returns the stacklevel value for warnings.warn() for when the warning 

84 gets emitted by an imported module, but the warning should point at the 

85 code doing the import. 

86 

87 Pass import_hook=True if the warning gets generated by an import hook 

88 (warn() gets called in load_module(), see PEP302) 

89 """ 

90 

91 py_version = sys.version_info[:2] 

92 if py_version <= (3, 2): 

93 # 2.7 included 

94 return 4 if import_hook else 2 

95 elif py_version == (3, 3): 

96 return 8 if import_hook else 10 

97 elif py_version == (3, 4): 

98 return 10 if import_hook else 8 

99 else: 

100 # fixed again in 3.5+, see https://bugs.python.org/issue24305 

101 return 4 if import_hook else 2

104class DynamicImporter(object): 

105 

106 # Note: see PEP302 for the Importer Protocol implemented below. 

107 

108 def __init__(self, path): 

109 self.path = path 

110 

111 def find_module(self, fullname, path=None): 

112 if not fullname.startswith(self.path): 

113 return 

114 

115 path, namespace = fullname.rsplit('.', 1) 

116 if path != self.path: 

117 return 

118 

119 return self 

120 

121 def load_module(self, fullname): 

122 if fullname in sys.modules: 

123 return sys.modules[fullname] 

124 

125 path, namespace = fullname.rsplit('.', 1) 

126 

127 # is_registered() is faster than enumerate_versions() and 

128 # in the common case of a namespace getting loaded before its 

129 # dependencies, is_registered() returns True for all dependencies. 

130 if not repository.is_registered(namespace) and not \ 

131 repository.enumerate_versions(namespace): 

132 raise ImportError('cannot import name %s, ' 

133 'introspection typelib not found' % namespace) 

134 

135 stacklevel = get_import_stacklevel(import_hook=True) 

136 with _check_require_version(namespace, stacklevel=stacklevel): 

137 try: 

138 introspection_module = get_introspection_module(namespace) 

139 except RepositoryError as e: 

140 raise ImportError(e) 

141 # Import all dependencies first so their init functions 

142 # (gdk_init, ..) in overrides get called. 

143 # https://bugzilla.gnome.org/show_bug.cgi?id=656314 

144 for dep in repository.get_immediate_dependencies(namespace): 

145 importlib.import_module('gi.repository.' + dep.split("-")[0]) 

146 dynamic_module = load_overrides(introspection_module) 

147 

148 dynamic_module.__file__ = '<%s>' % fullname 

149 dynamic_module.__loader__ = self 

150 sys.modules[fullname] = dynamic_module 

151 

152 return dynamic_module
