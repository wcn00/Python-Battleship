4def _playsoundWin(sound, block = True): 

5 ''' 

6 Utilizes windll.winmm. Tested and known to work with MP3 and WAVE on 

7 Windows 7 with Python 2.7. Probably works with more file formats. 

8 Probably works on Windows XP thru Windows 10. Probably works with all 

9 versions of Python. 

10 

11 Inspired by (but not copied from) Michael Gundlach <gundlach@gmail.com>'s mp3play: 

12 https://github.com/michaelgundlach/mp3play 

13 

14 I never would have tried using windll.winmm without seeing his code. 

15 ''' 

16 from ctypes import c_buffer, windll 

17 from random import random 

18 from time import sleep 

19 from sys import getfilesystemencoding 

20 

21 def winCommand(*command): 

22 buf = c_buffer(255) 

23 command = ' '.join(command).encode(getfilesystemencoding()) 

24 errorCode = int(windll.winmm.mciSendStringA(command, buf, 254, 0)) 

25 if errorCode: 

26 errorBuffer = c_buffer(255) 

27 windll.winmm.mciGetErrorStringA(errorCode, errorBuffer, 254) 

28 exceptionMessage = ('\n Error ' + str(errorCode) + ' for command:' 

29 '\n ' + command.decode() + 

30 '\n ' + errorBuffer.value.decode()) 

31 raise PlaysoundException(exceptionMessage) 

32 return buf.value 

33 

34 alias = 'playsound_' + str(random()) 

35 winCommand('open "' + sound + '" alias', alias) 

36 winCommand('set', alias, 'time format milliseconds') 

37 durationInMS = winCommand('status', alias, 'length') 

38 winCommand('play', alias, 'from 0 to', durationInMS.decode()) 

39 

40 if block: 

41 sleep(float(durationInMS) / 1000.0)
